Bit twidding:
    flag variables
    each int var is 32 bitwise
    use single int variable to strore multiple flags

#define JUMPING (1 << 0) // 0001 --> move the byte
#define RUNNING (1 << 1) // 0010
#define RUNNING (1 << 2) // 0100

int state = 0;
state |= JUMPING;
state &= ~JUMPING;

(then we use if else condition)

if (state & JUMPING) {
    ....
